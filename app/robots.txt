import { MetadataRoute } from "next";

export default function robots(): MetadataRoute.Robots {
  const baseUrl = "https://www.bouwnce.com";

  return {
    rules: [
      {
        userAgent: "*",
        allow: ["/", "/marketplace", "/waitlist", "/app/profile/"],
        disallow: [
          "/api/",
          "/admin/",
          "/app/",
          "/vendor/",
          "/login",
          "/register",
          "/*?*",
        ],
      },
    ],
    sitemap: `${baseUrl}/sitemap.xml`,
  };
}
